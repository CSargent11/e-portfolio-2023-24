A Class is like an object constructor, or a "blueprint" for creating objects. Within these there are different functions to help the programme run smoothly.

class Surfboard:
	
    
    def __init__(self, brand, length, width):
	
    
        self.brand = brand  # Unprotected variable
	
    
        self._length = length  # Protected variable
	
    
        self._width = width  # Protected variable
	
    
    def get_length(self):
	
    
        return self._length
	
    
    def set_length(self, length):
	
    
        self._length = length
	
    

    def get_width(self):
	
    
        return self._width
	
    

    def set_width(self, width):
	
    
        self._width = width
	
    
		
my_surfboard = Surfboard("ABC Surf Co.", 6.2, 22.5)
	
    
print("Brand:", my_surfboard.brand)  # Accessing unprotected variable
	
    
print("Length:", my_surfboard.get_length())  # Accessing protected variable using a getter method
	
    
print("Width:", my_surfboard.get_width())  # Accessing protected variable using a getter method
	
    

my_surfboard.brand = "XYZ Surfboards"  # Modifying unprotected variable
	
    
my_surfboard.set_length(6.4)  # Modifying protected variable using a setter method
	
    
my_surfboard.set_width(23.0)  # Modifying protected variable using a setter method
	
    

print("Updated Brand:", my_surfboard.brand)
	
    
print("Updated Length:", my_surfboard.get_length())
	
    
print("Updated Width:", my_surfboard.get_width())
